102:     def each_table_row(table, row_count, &block)
103:       if table_has_primary_column?(table) && row_count > (chunk_size = table_chunk_size(table))
104:         # adapted from ActiveRecord::Batches
105:         primary_key = table_primary_key(table)
106:         quoted_primary_key = "#{quote_table_name(table)}.#{quote_column_name(primary_key)}"
107:         select_where_primary_key =
108:           "SELECT * FROM #{quote_table_name(table)}" +
109:             " WHERE #{quoted_primary_key} %s" +
110:             " ORDER BY #{quoted_primary_key} ASC" +
111:             " LIMIT #{chunk_size}"
112:         rows = select_all_by_sql(select_where_primary_key % '>= 0')
113:         until rows.blank?
114:           rows.each(&block)
115:           break if rows.length < chunk_size
116:           rows = select_all_by_sql(select_where_primary_key % "> #{rows.last[primary_key].to_i}")
117:         end
118:       else
119:         table_rows(table).each(&block)
120:       end
121:     end

77:         case column.type
