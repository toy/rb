load_lib libffi-dg.exp

dg-init libffi-init

global srcdir subdir

global cxx_options

set cxx_options “ -shared-libgcc -lstdc++“

dg-runtest [lsort [glob -nocomplain $srcdir/$subdir/*.cc]] $cxx_options “-O0 -W -Wall” dg-runtest [lsort [glob -nocomplain $srcdir/$subdir/*.cc]] $cxx_options “-O2” dg-runtest [lsort [glob -nocomplain $srcdir/$subdir/*.cc]] $cxx_options “-O3” dg-runtest [lsort [glob -nocomplain $srcdir/$subdir/*.cc]] $cxx_options “-Os“

dg-finish
