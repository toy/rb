150:       line = sprintf(@format, *arguments)
151: 
152:       width = get_width
153:       if line.length == width - 1

155:       elsif line.length >= width

157:         if @bar_width == 0 then @out.print(line + eol) else show end
158:       else # line.length < width - 1

163:     end
