Ruby Net::GitHub::Upload

  • github.com/Constellation/ruby-net-github-upload
  • rubygems.org/gems/net-github-upload

DESCRIPTION

Ruby Net::GitHub::Upload is upload user agent for GitHub Downloads respect to github.com/typester/net-github-upload-perl

SYNOPSIS

  # require
  require 'rubygems'
  require 'net/github-upload'

  # setup
  login = `git config github.user`.chomp  # your login for github
  token = `git config github.token`.chomp # your token for github
  repos = 'repository name'               # your repos name (like 'taberareloo')
  gh = Net::GitHub::Upload.new(
    :login => login,
    :token => token
  )

  # file upload
  direct_link = gh.upload(
    :repos => repos,
    :file  => 'test/test',
    :description => "test file",
    :upload_timeout => 600 # Allow yourself time to upload larger files.
  )
  # direct link is link to Amazon S3.
  # Because GitHub refrection for file changing is async,
  # if you get changed file synchronously, you use this "direct_link"

  # data upload
  # you can define content_type => Amazon S3 Content-Type
  time = Time.now.to_i
  direct_link = gh.upload(
    :repos => repos,
    :data  => 'test',
    :name  => "test_#{time}.txt",
    :content_type => 'text/plain',
    :description => "test file2"
  )

  # replace file or data
  # thx id:rngtng !
  direct_link = gh.replace(
    :repos => repos,
    :file  => 'test/test',
    :description => "test file"
  )
  direct_link = gh.replace(
    :repos => repos,
    :data  => 'test',
    :name  => "test_#{time}.txt",
    :content_type => 'text/plain',
    :description => "test file2"
  )

  # Printing out progress (will yield to block once a second):
  direct_link = gh.upload(
    :repos => repos,
    :file  => 'test/test',
    :description => "test file",
    :upload_timeout => 600 # Allow yourself time to upload larger files.
  ) do
    print "."
  end
  puts "."
