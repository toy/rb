     # File lib/execjs/external_runtime.rb, line 124
124:       def locate_executable(cmd)
125:         if ExecJS.windows? && File.extname(cmd) == ""
126:           cmd << ".exe"
127:         end
128: 
129:         if File.executable? cmd
130:           cmd
131:         else
132:           path = ENV['PATH'].split(File::PATH_SEPARATOR).find { |p|
133:             full_path = File.join(p, cmd)
134:             File.executable?(full_path) && File.file?(full_path)
135:           }
136:           path && File.expand_path(cmd, path)
