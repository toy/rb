170:     @wbuffer = "" unless defined? @wbuffer
171:     @wbuffer << s
172:     @sync ||= false
173:     if @sync or @wbuffer.size > BLOCK_SIZE or idx = @wbuffer.rindex($/)
174:       remain = idx ? idx + $/.size : @wbuffer.length
175:       nwritten = 0
176:       while remain > 0

178:         begin
