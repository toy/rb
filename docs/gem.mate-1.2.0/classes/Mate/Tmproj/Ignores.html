50:     def process(parent, ignore_file)
51:       return unless ignore_file.exist?
52:       current_file_pattern = []
53:       current_folder_pattern = []
54:       ignore_file.readlines.map do |line|
55:         line.lstrip[/(.*?)(\r?\n)?$/, 1] # this strange stuff strips line, but allows mac Icon\r files to be ignored
56:       end.reject do |line|
57:         line.empty? || %w[# !].include?(line[0, 1])
58:       end.each do |line|
59:         pattern = Regexp.escape(line).gsub(/\\\*+/, '[^/]*') # understand only * pattern
60:         pattern = "#{DOUBLE_ASTERISK_R}#{pattern}" unless line['/']
61:         pattern.sub!(/^\//, '')
62:         unless pattern.sub!(/\/$/, '')
63:           current_file_pattern << pattern
64:         end
65:         current_folder_pattern << pattern
66:       end
67:       current_file_pattern = current_file_pattern.join('|')
68:       current_folder_pattern = current_folder_pattern.join('|')
69:       unless parent == dir
70:         parent_pattern = Regexp.escape(parent.relative_path_from(dir))
71:         current_file_pattern = "#{parent_pattern}/(?:#{current_file_pattern})"
72:         current_folder_pattern = "#{parent_pattern}/(?:#{current_folder_pattern})"
73:       end
74:       file_pattern << current_file_pattern
75:       folder_pattern << current_folder_pattern
76:     end
