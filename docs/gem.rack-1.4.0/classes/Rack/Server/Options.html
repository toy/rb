 96:       def handler_opts(options)
 97:         begin
 98:           info = []
 99:           server = Rack::Handler.get(options[:server]) || Rack::Handler.default(options)
100:           if server && server.respond_to?(:valid_options)
101:             info << ""
102:             info << "Server-specific options for #{server.name}:"
103: 
104:             has_options = false
105:             server.valid_options.each do |name, description|
106:               next if name.to_s.match(/^(Host|Port)[^a-zA-Z]/) # ignore handler's host and port options, we do our own.
107:               info << "  -O %-21s %s" % [name, description]
108:               has_options = true
109:             end
110:             return "" if !has_options
111:           end
112:           info.join("\n")
113:         rescue NameError
114:           return "Warning: Could not find handler specified (#{options[:server] || 'default'}) to determine handler-specific options"
115:         end
116:       end
