668:       type, data = API.QueryValue(@hkey, name)
669:       unless rtype.empty? or rtype.include?(type)
670:         raise TypeError, "Type mismatch (expect #{rtype.inspect} but #{type} present)"
671:       end
672:       case type
673:       when REG_SZ, REG_EXPAND_SZ

688:     end
