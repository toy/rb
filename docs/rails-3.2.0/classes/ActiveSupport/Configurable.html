Reads and writes attributes from a configuration OrderedHash. 

  require 'active_support/configurable'

  class User
    include ActiveSupport::Configurable
  end

  user = User.new

  user.config.allowed_access = true
  user.config.level = 1

  user.config.allowed_access # => true
  user.config.level          # => 1
