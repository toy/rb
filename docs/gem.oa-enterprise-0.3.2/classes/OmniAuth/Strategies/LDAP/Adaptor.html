 76:           connect(options) unless connecting?
 77:           begin
 78:           @bind_tried = true
 79: 
 80:           bind_dn = (options[:bind_dn] || @bind_dn).to_s
 81:           try_sasl = options.has_key?(:try_sasl) ? options[:try_sasl] : @try_sasl
 82:           if options.has_key?(:allow_anonymous)
 83:             allow_anonymous = options[:allow_anonymous]
 84:           else
 85:             allow_anonymous = @allow_anonymous
 86:           end
 87:           # Rough bind loop:
 88:           # Attempt 1: SASL if available
 89:           # Attempt 2: SIMPLE with credentials if password block
 90:           # Attempt 3: SIMPLE ANONYMOUS if 1 and 2 fail and allow anonymous is set to true
 91:           if try_sasl and sasl_bind(bind_dn, options)
 92:             puts "bound with sasl"
 93:           elsif simple_bind(bind_dn, options)
 94:             puts "bound with simple"
 95:           elsif allow_anonymous and bind_as_anonymous(options)
 96:             puts "bound as anonymous"
 97:           else
 98:             message = yield if block_given?
 99:             message ||= ('All authentication methods for %s exhausted.') % target
100:             raise AuthenticationError, message
101:           end
102:           @bound = true
103:           rescue Net::LDAP::LdapError
104:             raise AuthenticationError, $!.message
105:           end
106:         end
