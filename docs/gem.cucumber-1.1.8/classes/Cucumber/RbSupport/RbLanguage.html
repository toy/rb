81:         snippet_pattern = Regexp.escape(step_name).gsub('\ ', ' ').gsub('/', '\/')
82:         arg_count = 0

84:           snippet_pattern = snippet_pattern.gsub(Regexp.new(pattern), pattern)
85:           arg_count += snippet_pattern.scan(pattern).length
86:         end
87:

89:         block_args << multiline_arg_class.default_arg_name unless multiline_arg_class.nil?
90:         block_arg_string = block_args.empty? ? "" : " |#{block_args.join(", ")}|"
91:         multiline_class_comment = ""
92:         if(multiline_arg_class == Ast::Table)

94:         end
95:
