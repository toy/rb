117:         classname = "#{@feature_name}.#{@scenario}"
118:         name = "#{@scenario}#{suffix}"

121: 
122:         @builder.testcase(:classname => classname, :name => name, :time => "%.6f" % duration) do
123:           unless passed

125:               @builder.cdata! @output

127:             end

133:           end
134:         end
