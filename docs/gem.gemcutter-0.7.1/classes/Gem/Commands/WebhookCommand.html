     # File lib/rubygems/commands/webhook_command.rb, line 77
 77:   def list_webhooks
 78:     resp = rubygems_api_request(:get, "api/v1/web_hooks.yaml") do |request|
 79:       request.add_field("Authorization", Gem.configuration.rubygems_api_key)
 80:     end
 81: 
 82:     with_response(resp) do |response|
 83:       begin
 84:         groups = YAML.load(response.body)
 85: 
 86:         if groups.size.zero?
 87:           say "You haven't added any webhooks yet."
 88:         else
 89:           groups.each do |group, hooks|
 90:             if options[:global]
 91:               next if group != "all gems"
 92:             elsif options[:args] && options[:args].first
 93:               next if group != options[:args].first
 94:             end
 95: 
 96:             say "#{group}:"
 97:             hooks.each do |hook|
 98:               say "- #{hook['url']}"
 99:             end
100:           end
101:         end
102:       rescue Exception => error
103:         say "There was a problem parsing the data:"
104:         say error.to_s
105:         terminate_interaction
106:       end
107:     end
108:   end
