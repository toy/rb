42:       a, b = parts1.first, parts2.first
43:       case
44:         when a.nil? && b.nil? then 0
45:         when a.nil? then b.is_a?(Fixnum) ?  -1 : 1

49:         when b.is_a?(String) then -compare(parts2, parts1)

52:       end
