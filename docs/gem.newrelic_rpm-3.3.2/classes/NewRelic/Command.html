53:

55: 
56:       extra = []
57:       options = ARGV.options do |opts|
58:         script_name = File.basename($0)
59:         if script_name =~ /newrelic_cmd$/
60:           $stdout.puts "warning: the 'newrelic_cmd' script has been renamed 'newrelic'"
61:           script_name = 'newrelic'
62:         end
63:         opts.banner = "Usage: #{script_name} [ #{ @command_names.join(" | ")} ] [options]"
64:         opts.separator "use '#{script_name} <command> -h' to see detailed command options"
65:         opts
66:       end
67:       extra = options.order!
68:       command = extra.shift
69:       # just make it a little easier on them
70:       command = 'deployments' if command =~ /deploy/
71:       if command.nil?
72:         STDERR.puts options
73:       elsif !@command_names.include?(command)
74:         STDERR.puts "Unrecognized command: #{command}"
75:         STDERR.puts options
76:       else
77:         command_class = @commands.find{ |c| c.command == command}
78:         command_class.new(extra).run
79:       end
