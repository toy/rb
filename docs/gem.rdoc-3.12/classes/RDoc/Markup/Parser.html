Class RDoc::Markup::Parser < Object

  • lib/rdoc/markup/parser.rb

A recursive-descent parser for RDoc markup.

The parser tokenizes an input string then parses the tokens into a Document. Documents can be converted into output formats by writing a visitor like RDoc::Markup::ToHTML.

The parser only handles the block-level constructs Paragraph, List, ListItem, Heading, Verbatim, BlankLine and Rule. Inline markup such as +blah+ is handled separately by RDoc::Markup::AttributeManager.

To see what markup the Parser implements read RDoc. To see how to use RDoc markup to format text in your program read RDoc::Markup.

build_verbatim(margin)

Builds a Verbatim that is indented from margin.

The verbatim block is shifted left (the least indented lines start in column 0). Each part of the verbatim is one line of text, always terminated by a newline. Blank lines always consist of a single newline character, and there is never a single newline at the end of the verbatim.

232:     p :verbatim_begin => margin if @debug
233:     verbatim = RDoc::Markup::Verbatim.new
234: 
235:     min_indent = nil
236:     generate_leading_spaces = true
237:     line = ''
238: 
239:     until @tokens.empty? do
240:       type, data, column, = get
241: 
242:       if type == :NEWLINE then
243:         line << data
244:         verbatim << line
245:         line = ''
246:         generate_leading_spaces = true
247:         next
248:       end
249: 
250:       if column <= margin

252:         break
253:       end
254: 
255:       if generate_leading_spaces then

parse(parent, indent = 0)

Parses the tokens into an array of RDoc::Markup::XXX objects, and appends them to the passed parent RDoc::Markup::YYY object.

Exits at the end of the token stream, or when it encounters a token in a column less than indent (unless it is a NEWLINE).

Returns parent.

Turns text input into a stream of tokens

Source: show 

     # File lib/rdoc/markup/parser.rb, line 402
402:   def tokenize input
403:     s = StringScanner.new input
404: 
405:     @line = 0
406:     @line_pos = 0
407: 
408:     until s.eos? do
409:       pos = s.pos
410: 
411:       # leading spaces will be reflected by the column of the next token
412:       # the only thing we loose are trailing spaces at the end of the file
413:       next if s.scan(/ +/)
414: 
415:       # note: after BULLET, LABEL, etc.,
416:       # indent will be the column of the next non-newline token
417: 
418:       @tokens << case
419:                  # [CR]LF => :NEWLINE
420:                  when s.scan(/\r?\n/) then
421:                    token = [:NEWLINE, s.matched, *token_pos(pos)]
422:                    @line_pos = s.pos
423:                    @line += 1
424:                    token
425:                  # === text => :HEADER then :TEXT
426:                  when s.scan(/(=+)(\s*)/) then
427:                    level = s[1].length
428:                    header = [:HEADER, level, *token_pos(pos)]
429: 
430:                    if s[2] =~ /^\r?\n/ then
431:                      s.pos -= s[2].length
432:                      header
433:                    else
434:                      pos = s.pos
435:                      s.scan(/.*/)
436:                      @tokens << header
437:                      [:TEXT, s.matched.sub(/\r$/, ''), *token_pos(pos)]
438:                    end
439:                  # --- (at least 3) and nothing else on the line => :RULE
440:                  when s.scan(/(-{3,}) *\r?$/) then
441:                    [:RULE, s[1].length - 2, *token_pos(pos)]
442:                  # * or - followed by white space and text => :BULLET
443:                  when s.scan(/([*-]) +(\S)/) then
444:                    s.pos -= s[2].bytesize # unget \S
445:                    [:BULLET, s[1], *token_pos(pos)]
446:                  # A. text, a. text, 12. text => :UALPHA, :LALPHA, :NUMBER
447:                  when s.scan(/([a-z]|\d+)\. +(\S)/i) then
448:                    # FIXME if tab(s), the column will be wrong
449:                    # either support tabs everywhere by first expanding them to
450:                    # spaces, or assume that they will have been replaced
451:                    # before (and provide a check for that at least in debug
452:                    # mode)
453:                    list_label = s[1]
454:                    s.pos -= s[2].bytesize # unget \S
455:                    list_type =
456:                      case list_label
457:                      when /[a-z]/ then :LALPHA
458:                      when /[A-Z]/ then :UALPHA
459:                      when /\d/    then :NUMBER
460:                      else
461:                        raise ParseError, "BUG token #{list_label}"
462:                      end
463:                    [list_type, list_label, *token_pos(pos)]
464:                  # [text] followed by spaces or end of line => :LABEL
465:                  when s.scan(/\[(.*?)\]( +|\r?$)/) then
466:                    [:LABEL, s[1], *token_pos(pos)]
467:                  # text:: followed by spaces or end of line => :NOTE
468:                  when s.scan(/(.*?)::( +|\r?$)/) then
469:                    [:NOTE, s[1], *token_pos(pos)]
470:                  # anything else: :TEXT
471:                  else s.scan(/(.*?)(  )?\r?$/)
472:                    token = [:TEXT, s[1], *token_pos(pos)]
473: 
474:                    if s[2] then
475:                      @tokens << token
476:                      [:BREAK, s[2], *token_pos(pos + s[1].length)]
477:                    else
478:                      token
479:                    end
480:                  end
481:     end
