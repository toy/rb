128:             def rgb_to_hsv(r, g, b)
129:                h, s, v = 0.0, 0.0, 0.0
130:       
131:                _r = r.convert_base(16, 10).to_f / 255.0
132:                _g = g.convert_base(16, 10).to_f / 255.0
133:                _b = b.convert_base(16, 10).to_f / 255.0
134:                rgb = [ _r, _g, _b ]
135:       
136:                min = rgb.min
137:                max = rgb.max
138:                v = max
139:       
140:                delta = max - min
141:                if max != 0.0
142:                   s = delta / max
143:                else
144:                   return [-1, 0, v]
145:                end
146:       
147:                if _r == max
148:                   h = ( _g - _b ) / delta
149:                elsif( _g == max )
150:                   h = 2 + ( _b - _r ) / delta
151:                else
152:                   h = 4 + ( _r - _g ) / delta
153:                end
