Class ActiveRecord::ConnectionAdapters::SQLiteAdapter < AbstractAdapter

  • activerecord-3.1.3/lib/active_record/connection_adapters/sqlite_adapter.rb

The SQLite adapter works with both the 2.x and 3.x series of SQLite with the sqlite-ruby drivers (available both as gems and from rubyforge.org/projects/sqlite-ruby/).

Options:

  • :database - Path to the database file.

     # File activerecord-3.1.3/lib/active_record/connection_adapters/sqlite_adapter.rb, line 222
222:       def exec_query(sql, name = nil, binds = [])
223:         log(sql, name, binds) do
224: 
225:           # Don't cache statements without bind values
226:           if binds.empty?
227:             stmt    = @connection.prepare(sql)
228:             cols    = stmt.columns
229:             records = stmt.to_a
230:             stmt.close
231:             stmt = records
232:           else
233:             cache = @statements[sql] ||= {
234:               :stmt => @connection.prepare(sql)
235:             }
236:             stmt = cache[:stmt]
237:             cols = cache[:cols] ||= stmt.columns
238:             stmt.reset!
239:             stmt.bind_params binds.map { |col, val|
240:               type_cast(val, col)
241:             }
242:           end
243: 
244:           ActiveRecord::Result.new(cols, stmt.to_a)
245:         end
246:       end
