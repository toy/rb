621:       def build_query_multipart_str(query, boundary)
622:         parts = Parts.new
623:         query.each do |attr, value|
624:           value ||= ''
625:           headers = ["--#{boundary}"]
626:           if Message.file?(value)
627:             remember_pos(value)
628:             param_str = params_from_file(value).collect { |k, v|
629:               "#{k}=\"#{v}\""
630:             }.join("; ")
631:             if value.respond_to?(:mime_type)
632:               content_type = value.mime_type
633:             elsif value.respond_to?(:content_type)
634:               content_type = value.content_type
635:             else
636:               path = value.respond_to?(:path) ? value.path : nil
637:               content_type = Message.mime_type(path)
638:             end
639:             headers << %{Content-Disposition: form-data; name="#{attr}"; #{param_str}}
640:             headers << %{Content-Type: #{content_type}}
641:           elsif attr.is_a?(Hash)
642:             h = attr
643:             value = h[:content]
644:             h.each do |h_key, h_val|
645:               headers << %{#{h_key}: #{h_val}} if h_key != :content
646:             end
647:             remember_pos(value) if Message.file?(value)
648:           else
649:             headers << %{Content-Disposition: form-data; name="#{attr}"}
650:           end
651:           parts.add(headers.join(CRLF) + CRLF + CRLF)
652:           parts.add(value)
653:           parts.add(CRLF)
654:         end
655:         parts.add("--#{boundary}--" + CRLF + CRLF) # empty epilogue
656:         parts
657:       end
