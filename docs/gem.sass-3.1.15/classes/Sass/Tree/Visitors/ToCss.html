24:     result = String.new
25:     node.children.each do |child|
26:       next if child.invisible?
27:       child_str = visit(child)
28:       result << child_str + (node.style == :compressed ? '' : "\n")
29:     end
30:     result.rstrip!
31:     return "" if result.empty?
32:     result << "\n"
33:     unless Sass::Util.ruby1_8? || result.ascii_only?
34:       if node.children.first.is_a?(Sass::Tree::CharsetNode)
35:         begin
36:           encoding = node.children.first.name
37:           # Default to big-endian encoding, because we have to decide somehow
38:           encoding << 'BE' if encoding =~ /\Autf-(16|32)\Z/i
39:           result = result.encode(Encoding.find(encoding))
40:         rescue EncodingError
41:         end
42:       end

46:       }".encode(result.encoding) + result

125:   def visit_rule(node)
126:     with_tabs(@tabs + node.tabs) do
127:       rule_separator = node.style == :compressed ? ',' : ', '
128:       line_separator =
129:         case node.style
130:           when :nested, :expanded; "\n"
131:           when :compressed; ""
132:           else; " "
133:         end
134:       rule_indent = '  ' * @tabs
135:       per_rule_indent, total_indent = [:nested, :expanded].include?(node.style) ? [rule_indent, ''] : ['', rule_indent]
136: 
137:       joined_rules = node.resolved_rules.members.map do |seq|
138:         rule_part = seq.to_a.join
139:         if node.style == :compressed
140:           rule_part.gsub!(/([^,])\s*\n\s*/m, '\1 ')
141:           rule_part.gsub!(/\s*([,+>])\s*/m, '\1')
142:           rule_part.strip!
143:         end
144:         rule_part
145:       end.join(rule_separator)
146: 
147:       joined_rules.sub!(/\A\s*/, per_rule_indent)
148:       joined_rules.gsub!(/\s*\n\s*/, "#{line_separator}#{per_rule_indent}")
149:       total_rule = total_indent << joined_rules
150: 
151:       to_return = ''
152:       old_spaces = '  ' * @tabs
153:       spaces = '  ' * (@tabs + 1)
154:       if node.style != :compressed
155:         if node.options[:debug_info] && !@in_directive
156:           to_return << visit(debug_info_rule(node.debug_info, node.options)) << "\n"
157:         elsif node.options[:trace_selectors]
158:           to_return << "#{old_spaces}/* "
159:           to_return << node.stack_trace.join("\n   #{old_spaces}")
160:           to_return << " */\n"
161:         elsif node.options[:line_comments]
162:           to_return << "#{old_spaces}/* line #{node.line}"
163: 
164:           if node.filename
165:             relative_filename = if node.options[:css_filename]
166:               begin
167:                 Pathname.new(node.filename).relative_path_from(
168:                   Pathname.new(File.dirname(node.options[:css_filename]))).to_s
169:               rescue ArgumentError
170:                 nil
171:               end
172:             end
173:             relative_filename ||= node.filename
174:             to_return << ", #{relative_filename}"
175:           end
176: 
177:           to_return << " */\n"
178:         end
179:       end
180: 
181:       if node.style == :compact
182:         properties = with_tabs(0) {node.children.map {|a| visit(a)}.join(' ')}
183:         to_return << "#{total_rule} { #{properties} }#{"\n" if node.group_end}"
184:       elsif node.style == :compressed
185:         properties = with_tabs(0) {node.children.map {|a| visit(a)}.join(';')}
186:         to_return << "#{total_rule}{#{properties}}"
187:       else
188:         properties = with_tabs(@tabs + 1) {node.children.map {|a| visit(a)}.join("\n")}
189:         end_props = (node.style == :expanded ? "\n" + old_spaces : ' ')
190:         to_return << "#{total_rule} {\n#{properties}#{end_props}}#{"\n" if node.group_end}"
191:       end
192: 
193:       to_return
194:     end
195:   end
