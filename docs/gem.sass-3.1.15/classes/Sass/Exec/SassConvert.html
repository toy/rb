632:         end
633:

644:         @options[:for_engine][:syntax] = @options[:from]

648:             if @options[:from] == :css
649:               require 'sass/css'
650:               ::Sass::CSS.new(input.read, @options[:for_tree]).render(@options[:to])
651:             elsif @options[:from] == :less
