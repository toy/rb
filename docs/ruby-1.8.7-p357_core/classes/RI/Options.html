267: 
268:           when "--doc-dir"
269:             if File.directory?(arg)
270:               @doc_dirs << arg
271:             else
272:               $stderr.puts "Invalid directory: #{arg}"
273:               exit 1
274:             end
275: 
276:           when "--format"

278:             unless @formatter
279:               $stderr.print "Invalid formatter (should be one of "

282:             end
283:           when "--width"
284:             begin
285:               @width = Integer(arg)

288:               exit 1
289:             end

291:         end
