569:     def update_attributes(current, wanted)
570:       str = ""
571:       # first turn off unwanted ones
572:       off = current & ~wanted
573:       for quality in [ BOLD, ITALIC, CODE]
574:         if (off & quality) > 0
575:           str << "</" + ATTR_MAP[quality]

577:       end
578:
