21:       def upload info
22:         unless info[:repos]
23:           raise "required repository name"
24:         end
25:         info[:repos] = @login + '/' + info[:repos] unless info[:repos].include? '/'
26: 
27:         if info[:file]
28:           file = info[:file]
29:           unless File.exist?(file) && File.readable?(file)
30:             raise "file does not exsits or readable"
31:           end
32:           info[:name] ||= File.basename(file)
33:         end
34:         unless  info[:file] || info[:data]
35:           raise "required file or data parameter to upload"
36:         end
37: 
38:         unless info[:name]
39:           raise "required name parameter for filename with data parameter"
40:         end
41: 
42:         if info[:replace]
43:           list_files(info[:repos]).each { |obj|
44:             next unless obj[:name] == info[:name]
45:             delete info[:repos], obj[:id]
46:           }
47:         elsif list_files(info[:repos]).any?{|obj| obj[:name] == info[:name]}
48:           raise "file '#{info[:name]}' is already uploaded. please try different name"
49:         end
50: 
51:         info[:content_type] ||= 'application/octet-stream'
52:         stat = HTTPClient.post("https://github.com/#{info[:repos]}/downloads", {
53:           "file_size"    => info[:file] ? File.stat(info[:file]).size : info[:data].size,
54:           "content_type" => info[:content_type],
55:           "file_name"    => info[:name],
56:           "description"  => info[:description] || '',
57:           "login"        => @login,
58:           "token"        => @token
59:         })
60: 
61:         unless stat.code == 200
62:           raise "Failed to post file info"
63:         end
64: 
65:         upload_info = JSON.parse(stat.content)
66:         if info[:file]
67:           f = File.open(info[:file], 'rb')
68:         else
69:           f = Tempfile.open('net-github-upload')
70:           f << info[:data]
71:           f.flush
72:         end
73:         stat = HTTPClient.post("http://github.s3.amazonaws.com/", [
74:           ['Filename', info[:name]],
75:           ['policy', upload_info['policy']],
76:           ['success_action_status', 201],
77:           ['key', upload_info['path']],
78:           ['AWSAccessKeyId', upload_info['accesskeyid']],
79:           ['Content-Type', upload_info['content_type'] || 'application/octet-stream'],
80:           ['signature', upload_info['signature']],
81:           ['acl', upload_info['acl']],
82:           ['file', f]
83:         ])
84:         f.close
85: 
86:         if stat.code == 201
87:           return FasterXmlSimple.xml_in(stat.content)['PostResponse']['Location']
88:         else
89:           raise 'Failed to upload' + extract_error_message(stat)
90:         end
91:       end
