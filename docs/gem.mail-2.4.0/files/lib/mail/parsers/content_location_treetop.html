module Mail 

  grammar ContentLocation

    include RFC2822
    include RFC2045

    rule primary
      CFWS location CFWS
    end

    rule location
      quoted_string {
        def text_value
          quoted_content.text_value
        end
      } / (token / [\x3d])+
    end

  end

end
