58:     # used to pack object specifiers, etc.
59:     # pack key-value pairs into an AEListDesc, then coerce it to the desired type
60:     # (there are other AEM APIs for packing obj specs, but this way is easiest)
61:     desc = AE::AEDesc.new_list(true)
62:     lst.each { |key, value| desc.put_param(key, value) }
63:     return desc.coerce(type)
64:   end
