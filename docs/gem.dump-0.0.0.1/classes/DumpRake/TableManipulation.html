 96:     def each_table_row(table, row_count, &block)
 97:       if table_has_primary_column?(table) && row_count > (chunk_size = table_chunk_size(table))
 98:         # adapted from ActiveRecord::Batches
 99:         primary_key = table_primary_key(table)
100:         quoted_primary_key = "#{quote_table_name(table)}.#{quote_column_name(primary_key)}"
101:         select_where_primary_key =
102:           "SELECT * FROM #{quote_table_name(table)}" +
103:             " WHERE #{quoted_primary_key} %s" +
104:             " ORDER BY #{quoted_primary_key} ASC" +
105:             " LIMIT #{chunk_size}"
106:         rows = select_all_by_sql(select_where_primary_key % '>= 0')
107:         until rows.blank?
108:           rows.each(&block)
109:           break if rows.length < chunk_size
110:           rows = select_all_by_sql(select_where_primary_key % "> #{rows.last[primary_key].to_i}")
111:         end
112:       else
113:         table_rows(table).each(&block)
114:       end
115:     end

69:     def table_chunk_size(table)
70:       expected_row_size = table_columns(table).map do |column|
71:         case column.type
72:         when :text
73:           Math.sqrt(column.limit || 2_147_483_647)
74:         when :string
75:           Math.sqrt(column.limit || 255)
76:         else
77:           column.limit || 10
78:         end
79:       end.sum
80:       [[(10_000_000 / expected_row_size).round, CHUNK_SIZE_MIN].max, CHUNK_SIZE_MAX].min
81:     end
