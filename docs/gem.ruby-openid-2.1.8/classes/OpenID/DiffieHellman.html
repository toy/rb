71:     def DiffieHellman.powermod(x, n, q)
72:       counter=0
73:       n_p=n
74:       y_p=1
75:       z_p=x
76:       while n_p != 0
77:         if n_p[0]==1
78:           y_p=(y_p*z_p) % q
79:         end
80:         n_p = n_p >> 1
81:         z_p = (z_p * z_p) % q
82:         counter += 1
83:       end
84:       return y_p
85:     end
