199:           end

205:             conn.request_post(url.request_uri, body, headers)
206:           end
207:         }

211:       rescue RuntimeError => why

214:         raise SSLFetchingError, "Error connecting to SSL URL #{url}: #{why}"

218:         raise FetchingError, "Error fetching #{url}: #{why}"
