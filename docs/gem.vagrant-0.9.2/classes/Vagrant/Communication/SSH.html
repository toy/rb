 93:         if @connection && !@connection.closed?
 94:           # There is a chance that the socket is closed despite us checking
 95:           # 'closed?' above. To test this we need to send data through the
 96:           # socket.
 97:           begin
 98:             @connection.exec!("")
 99:           rescue IOError
100:             @logger.info("Connection has been closed. Not re-using.")
101:             @connection = nil
102:           end
103: 
104:           # If the @connection is still around, then it is valid,
105:           # and we use it.
106:           if @connection
107:             @logger.debug("Re-using SSH connection.")
108:             return yield @connection if block_given?
109:             return
110:           end
111:         end
112: 
113:         ssh_info = @vm.ssh.info
114: 
115:         # Build the options we'll use to initiate the connection via Net::SSH
116:         opts = {
117:           :port                  => ssh_info[:port],
118:           :keys                  => [ssh_info[:private_key_path]],
119:           :keys_only             => true,
120:           :user_known_hosts_file => [],
121:           :paranoid              => false,
122:           :config                => false,
123:           :forward_agent         => ssh_info[:forward_agent]
124:         }
125: 
126:         # Check that the private key permissions are valid
127:         @vm.ssh.check_key_permissions(ssh_info[:private_key_path])
128: 
129:         # Connect to SSH, giving it a few tries
130:         @logger.info("Connecting to SSH: #{ssh_info[:host]}:#{ssh_info[:port]}")
131:         exceptions = [Errno::ECONNREFUSED, Net::SSH::Disconnect]
132:         connection = retryable(:tries => @vm.config.ssh.max_tries, :on => exceptions) do
133:           Net::SSH.start(ssh_info[:host], ssh_info[:username], opts)
134:         end
135: 
136:         @connection = connection

139:         # Net::SSH is simply not robust enough to handle... see

156:       def shell_execute(connection, command, sudo=false)
157:         @logger.info("Execute: #{command} (sudo=#{sudo.inspect})")
158:         exit_status = nil
159: 
160:         # Determine the shell to execute. If we are using `sudo` then we
161:         # need to wrap the shell in a `sudo` call.
162:         shell = "#{@vm.config.ssh.shell} -l"
163:         shell = "sudo -H #{shell}" if sudo
164: 
165:         # Open the channel so we can execute or command
166:         channel = connection.open_channel do |ch|
167:           ch.exec(shell) do |ch2, _|
168:             # Setup the channel callbacks so we can get data and exit status
169:             ch2.on_data do |ch3, data|
170:               if block_given?
171:                 # Filter out the clear screen command
172:                 data.gsub!("\e[H", "")
173:                 @logger.debug("stdout: #{data}")
174:                 yield :stdout, data
175:               end
176:             end
177: 
178:             ch2.on_extended_data do |ch3, type, data|
179:               if block_given?
180:                 # Filter out the clear screen command
181:                 data.gsub!("\e[H", "")
182:                 @logger.debug("stderr: #{data}")
183:                 yield :stderr, data
184:               end
185:             end
186: 
187:             ch2.on_request("exit-status") do |ch3, data|
188:               exit_status = data.read_long
189:               @logger.debug("Exit status: #{exit_status}")
190:             end
191: 
192:             # Set the terminal
193:             ch2.send_data "export TERM=vt100\n"
194: 
195:             # Output the command
196:             ch2.send_data "#{command}\n"
197: 
198:             # Remember to exit or this channel will hang open
199:             ch2.send_data "exit\n"
200:           end
201:         end
202: 
203:         # Wait for the channel to complete
204:         channel.wait
205: 
206:         # Return the final exit status
207:         return exit_status
208:       end
