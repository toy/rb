sync_unlock(m = EX)

Source: show 

     # File lib/sync.rb, line 159
159:   def sync_unlock(m = EX)
160:     wakeup_threads = []
161:     @sync_mutex.synchronize do
162:       if sync_mode == UN
163:         Err::UnknownLocker.Fail(Thread.current)
164:       end
165: 
166:       m = sync_mode if m == EX and sync_mode == SH
167: 
168:       runnable = false
169:       case m
170:       when UN
171:         Err::UnknownLocker.Fail(Thread.current)
172: 
173:       when EX
174:         if sync_ex_locker == Thread.current
175:           if (self.sync_ex_count = sync_ex_count - 1) == 0
176:             self.sync_ex_locker = nil
177:             if sync_sh_locker.include?(Thread.current)
178:               self.sync_mode = SH
179:             else
180:               self.sync_mode = UN
181:             end
182:             runnable = true
183:           end
184:         else
185:           Err::UnknownLocker.Fail(Thread.current)
186:         end
187: 
188:       when SH
189:         if (count = sync_sh_locker[Thread.current]).nil?
190:           Err::UnknownLocker.Fail(Thread.current)
191:         else
192:           if (sync_sh_locker[Thread.current] = count - 1) == 0
193:             sync_sh_locker.delete(Thread.current)
194:             if sync_sh_locker.empty? and sync_ex_count == 0
195:               self.sync_mode = UN
196:               runnable = true
197:             end
198:           end
199:         end
200:       end
201: 
202:       if runnable
203:         if sync_upgrade_waiting.size > 0
204:           th, count = sync_upgrade_waiting.shift
205:           sync_sh_locker[th] = count
206:           th.wakeup
207:           wakeup_threads.push th
208:         else
209:           wait = sync_waiting
210:           self.sync_waiting = []
211:           for th in wait
212:             th.wakeup
213:             wakeup_threads.push th
214:           end
215:         end
216:       end
217:     end
218:     for th in wakeup_threads
219:       th.run
220:     end
221:     self
222:   end

     # File lib/sync.rb, line 264
264:   def sync_try_lock_sub(m)
265:     case m
266:     when SH
267:       case sync_mode
268:       when UN
269:         self.sync_mode = m
270:         sync_sh_locker[Thread.current] = 1
271:         ret = true
272:       when SH
273:         count = 0 unless count = sync_sh_locker[Thread.current]
274:         sync_sh_locker[Thread.current] = count + 1
275:         ret = true
276:       when EX
277:         # in EX mode, lock will upgrade to EX lock
278:         if sync_ex_locker == Thread.current
279:           self.sync_ex_count = sync_ex_count + 1
280:           ret = true
281:         else
282:           ret = false
283:         end
284:       end
285:     when EX
286:       if sync_mode == UN or
287:           sync_mode == SH && sync_sh_locker.size == 1 && sync_sh_locker.include?(Thread.current)
288:         self.sync_mode = m
289:         self.sync_ex_locker = Thread.current
290:         self.sync_ex_count = 1
291:         ret = true
292:       elsif sync_mode == EX && sync_ex_locker == Thread.current
293:         self.sync_ex_count = sync_ex_count + 1
294:         ret = true
295:       else
296:         ret = false
297:       end
298:     else
299:       Err::LockModeFailer.Fail mode
300:     end
301:     return ret
302:   end
