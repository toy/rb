Class RDoc::RI::Store < Object

  • lib/rdoc/ri/store.rb

A set of ri data.

The store manages reading and writing ri data for a project (gem, path, etc.) and maintains a cache of methods, classes and ancestors in the store.

The store maintains a cache of its contents for faster lookup. After adding items to the store it must be flushed using save_cache. The cache contains the following structures: 

   @cache = {
     :class_methods    => {}, # class name => class methods
     :instance_methods => {}, # class name => instance methods
     :attributes       => {}, # class name => attributes
     :modules          => [], # classes and modules in this store
     :ancestors        => {}, # class name => ancestor names
   }

     # File lib/rdoc/ri/store.rb, line 261
261:   def save_class klass
262:     full_name = klass.full_name
263: 
264:     FileUtils.mkdir_p class_path(full_name) unless @dry_run
265: 
266:     @cache[:modules] << full_name
267: 
268:     path = class_file full_name
269: 
270:     begin
271:       disk_klass = load_class full_name
272: 
273:       klass = disk_klass.merge klass
274:     rescue Errno::ENOENT
275:     end
276: 
277:     # BasicObject has no ancestors
278:     ancestors = klass.ancestors.compact.map do |ancestor|
279:       # HACK for classes we don't know about (class X < RuntimeError)
280:       String === ancestor ? ancestor : ancestor.full_name
281:     end
282: 
283:     @cache[:ancestors][full_name] ||= []
284:     @cache[:ancestors][full_name].push(*ancestors)
285: 
286:     attributes = klass.attributes.map do |attribute|
287:       "#{attribute.definition} #{attribute.name}"
288:     end
289: 
290:     unless attributes.empty? then
291:       @cache[:attributes][full_name] ||= []
292:       @cache[:attributes][full_name].push(*attributes)
293:     end
294: 
295:     to_delete = []
296: 
297:     unless klass.method_list.empty? then
298:       @cache[:class_methods][full_name]    ||= []
299:       @cache[:instance_methods][full_name] ||= []
300: 
301:       class_methods, instance_methods =
302:         klass.method_list.partition { |meth| meth.singleton }
303: 
304:       class_methods    = class_methods.   map { |method| method.name }
305:       instance_methods = instance_methods.map { |method| method.name }
306: 
307:       old = @cache[:class_methods][full_name] - class_methods
308:       to_delete.concat old.map { |method|
309:         method_file full_name, "#{full_name}::#{method}"
310:       }
311: 
312:       old = @cache[:instance_methods][full_name] - instance_methods
313:       to_delete.concat old.map { |method|
314:         method_file full_name, "#{full_name}##{method}"
315:       }
316: 
317:       @cache[:class_methods][full_name]    = class_methods
318:       @cache[:instance_methods][full_name] = instance_methods
319:     end
320: 
321:     return if @dry_run
322: 
323:     FileUtils.rm_f to_delete
324: 
325:     marshal = Marshal.dump klass
326: 
327:     open path, 'wb' do |io|
328:       io.write marshal
329:     end
330:   end
