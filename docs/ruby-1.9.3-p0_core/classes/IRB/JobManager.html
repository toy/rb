 86:     def delete(key)
 87:       case key
 88:       when Integer
 89:   IRB.fail NoSuchJob, key unless @jobs[key]
 90:   @jobs[key] = nil
 91:       else
 92:   catch(:EXISTS) do
 93:     @jobs.each_index do
 94:       |i|
 95:       if @jobs[i] and (@jobs[i][0] == key ||
 96:            @jobs[i][1] == key ||
 97:            @jobs[i][1].context.main.equal?(key))
 98:         @jobs[i] = nil
 99:         throw :EXISTS
100:       end
101:     end
102:     IRB.fail NoSuchJob, key
103:   end
104:       end
105:       until assoc = @jobs.pop; end unless @jobs.empty?
106:       @jobs.push assoc
107:     end

71:     def search(key)
72:       job = case key
73:       when Integer
74:   @jobs[key]
75:       when Irb
76:   @jobs.find{|k, v| v.equal?(key)}
77:       when Thread
78:   @jobs.assoc(key)
79:       else
80:   @jobs.find{|k, v| v.context.main.equal?(key)}
81:       end
82:       IRB.fail NoSuchJob, key if job.nil?
