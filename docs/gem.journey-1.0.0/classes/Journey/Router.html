118:       addr       = env['REMOTE_ADDR']
119:       req        = request_class.new env
120: 
121:       routes = filter_routes(env['PATH_INFO']) + custom_routes.find_all { |r|
122:         r.path.match(env['PATH_INFO'])
123:       }
124: 
125:       routes.sort_by(&:precedence).find_all { |r|
126:         r.constraints.all? { |k,v| v === req.send(k) } &&
127:           r.verb === env['REQUEST_METHOD']
128:       }.reject { |r| addr && !(r.ip === addr) }.map { |r|
129:         match_data  = r.path.match(env['PATH_INFO'])
130:         match_names = match_data.names.map { |n| n.to_sym }
131:         match_values = match_data.captures.map { |v| v && Utils.unescape_uri(v) }
132:         info = Hash[match_names.zip(match_values).find_all { |_,y| y }]
133:
