Class Diff::LCS::Hunk < Object

  • lib/diff/lcs/hunk.rb

A Hunk is a group of Blocks which overlap because of the context surrounding each block. (So if we’re not using context, every hunk will contain one block.) Used in the diff program (bin/diff).

111:     warn "Expecting only one block in an old diff hunk!" if @blocks.size > 1
112:     op_act = { "+" => 'a', "-" => 'd', "!" => "c" }
113:

116:       # Calculate item number range. Old diff range is just like a context
117:       # diff range, except the ranges are on one line with the action between
118:       # them.

Source: show 

     # File lib/diff/lcs/hunk.rb, line 128
128:   def unified_diff
129:       # Calculate item number range.
130:     s = "@@ -#{unified_range(:old)} +#{unified_range(:new)} @@\n"
131: 
132:       # Outlist starts containing the hunk of the old file. Removing an item
133:       # just means putting a '-' in front of it. Inserting an item requires
134:       # getting it from the new file and splicing it in. We splice in
135:       # +num_added+ items. Remove blocks use +num_added+ because splicing
136:       # changed the length of outlist.
137:       #
138:       # We remove +num_removed+ items. Insert blocks use +num_removed+
139:       # because their item numbers -- corresponding to positions in the NEW
140:       # file -- don't take removed items into account.
141:     lo, hi, num_added, num_removed = @start_old, @end_old, 0, 0
142: 
143:     outlist = @data_old[lo .. hi].collect { |e| e.gsub(/^/, ' ') }
144: 
145:     @blocks.each do |block|
146:       block.remove.each do |item|
147:         op = item.action.to_s # -
148:         offset = item.position - lo + num_added
149:         outlist[offset].gsub!(/^ /, op.to_s)
150:         num_removed += 1
151:       end
152:       block.insert.each do |item|
153:         op = item.action.to_s # +
154:         offset = item.position - @start_new + num_removed
155:         outlist[offset, 0] = "#{op}#{@data_new[item.position]}"
156:         num_added += 1
157:       end
158:     end
159: 
160:     s << outlist.join("\n")
161:   end
