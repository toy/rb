 52:     def run!(options = {})
 53:       Base.dry_run! if options[:dry_run]
 54:       Base.verbose_level = options[:verbose_level]
 55:       Progress.lines = true if Base.verbose_level >= 1
 56:       begin
 57:         read_config
 58:         tasks = Base.tasks(:update => update? || options[:update])
 59: 
 60:         if last_build_sdoc_version.nil? || last_build_sdoc_version != current_sdoc_version
 61:           puts "sdoc version changed - rebuilding all docs".red unless last_build_sdoc_version.nil?
 62:           Base.remove_if_present(Base.docs_path)
 63:         else
 64:           Base.chdir(Base.docs_path) do
 65:             to_delete = Dir.glob('*') - tasks.map(&:occupied_doc_pathes).flatten
 66:             to_delete.each do |path|
 67:               Base.remove_if_present(path)
 68:             end
 69:           end
 70:         end
 71: 
 72:         tasks.with_progress('sdoc').each do |task|
 73:           Progress.start(task.title) do
 74:             task.run(options)
 75:           end
 76:         end
 77: 
 78:         config_hash = Digest::SHA1.hexdigest(tasks.map(&:config_hash).inspect + title.to_s)
 79:         created_hash = config_hash_path.read rescue nil
 80: 
 81:         if config_hash != created_hash
 82:           Base.chdir(Base.docs_path) do
 83:             paths = []
 84:             titles = []
 85:             urls = []
 86:             tasks.each do |task|
 87:               doc_path = Base.docs_path + task.doc_path
 88:               if File.file?(doc_path + 'index.html')
 89:                 paths << task.doc_path
 90:                 titles << task.title
 91:                 urls << "docs/#{task.doc_path}"
 92:               end
 93:             end
 94: 
 95:             if paths.present?
 96:               Base.remove_if_present(Base.public_path)
 97: 
 98:               cmd = %w(sdoc-merge)
 99:               cmd << '-o' << Base.public_path
100:               cmd << '-t' << title
101:               cmd << '-n' << titles.join(',')
102:               cmd << '-u' << urls.join(' ')
103:               Base.system(*cmd + paths)
104: 
105:               if Base.public_path.directory?
106:                 (Base.public_path + 'docs').make_symlink(Base.docs_path.relative_path_from(Base.public_path))
107:                 config_hash_path.open('w') do |f|
108:                   f.write(config_hash)
109:                 end
110:                 last_build_sdoc_version_path.open('w') do |f|
111:                   f.write(current_sdoc_version)
112:                 end
113:               end
114:             end
115:           end
116:         end
117:       rescue ConfigError => e
118:         abort e.to_s.red.bold
119:       end
120:     end
