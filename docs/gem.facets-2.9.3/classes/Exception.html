 7:   def self.suppress(*exception_classes)
 8:     exception_classes.each do |e|
 9:       unless e < self
10:         raise ArgumentError, "exception #{e} not a subclass of #{self}"
11:       end
12:     end
13:     exception_classes = [self] | exception_classes

15:       yield
