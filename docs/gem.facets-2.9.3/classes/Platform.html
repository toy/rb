new(arch)

Source: show 

     # File lib/standard/facets/platform.rb, line 85
 85:   def initialize(arch)
 86:     case arch
 87:     when Array then
 88:       @cpu, @os, @version = arch
 89:     when String then
 90:       arch = arch.split '-'
 91: 
 92:       if arch.length > 2 and arch.last !~ /\d/ then # reassemble x86-linux-gnu
 93:         extra = arch.pop
 94:         arch.last << "-#{extra}"
 95:       end
 96: 
 97:       cpu = arch.shift
 98: 
 99:       @cpu = case cpu
100:              when /i\d86/ then 'x86'
101:              else cpu
102:              end
103: 
104:       if arch.length == 2 and arch.last =~ /^\d+$/ then # for command-line
105:         @os, @version = arch
106:         return
107:       end
108: 
109:       os, = arch
110:       @cpu, os = nil, cpu if os.nil? # legacy jruby
111: 
112:       @os, @version = case os
113:         when /aix(\d+)/ then             [ 'aix',       $1  ]
114:         when /cygwin/ then               [ 'cygwin',    nil ]
115:         when /darwin(\d+)?/ then         [ 'darwin',    $1  ]
116:         when /freebsd(\d+)/ then         [ 'freebsd',   $1  ]
117:         when /hpux(\d+)/ then            [ 'hpux',      $1  ]
118:         when /^java$/, /^jruby$/ then    [ 'java',      nil ]
119:         when /^java([\d.]*)/ then        [ 'java',      $1  ]
120:         when /linux/ then                [ 'linux',     $1  ]
121:         when /mingw32/ then              [ 'mingw32',   nil ]
122:         when /(mswin\d+)(\_(\d+))?/ then
123:           os, version = $1, $3
124:           @cpu = 'x86' if @cpu.nil? and os =~ /32$/
125:           [os, version]
126:         when /netbsdelf/ then            [ 'netbsdelf', nil ]
127:         when /openbsd(\d+\.\d+)/ then    [ 'openbsd',   $1  ]
128:         when /solaris(\d+\.\d+)/ then    [ 'solaris',   $1  ]
129:         # test
130:         when /^(\w+_platform)(\d+)/ then [ $1,          $2  ]
131:         else                             [ 'unknown',   nil ]
132:       end
133:     when Platform then
134:       @cpu = arch.cpu
135:       @os = arch.os
136:       @version = arch.version
137:     else
138:       raise ArgumentError, "invalid argument #{arch.inspect}"
139:     end
140:   end
