154:       def import(ovf)
155:         output = ""
156:         total = ""
157:         last  = 0
158:         execute("import", ovf) do |type, data|
159:           if type == :stdout
160:             # Keep track of the stdout so that we can get the VM name
161:             output << data
162:           elsif type == :stderr
163:             # Append the data so we can see the full view
164:             total << data
165: 
166:             # Break up the lines. We can't get the progress until we see an "OK"
167:             lines = total.split("\n")
168:             if lines.include?("OK.")
169:               # The progress of the import will be in the last line. Do a greedy
170:               # regular expression to find what we're looking for.
171:               if lines.last =~ /.+(\d{2})%/
172:                 current = $1.to_i
173:                 if current > last
174:                   last = current
175:                   yield current if block_given?
176:                 end
177:               end
178:             end
179:           end
180:         end
181: 
182:         # Find the name of the VM name
183:         if output !~ /Suggested VM name "(.+?)"/
184:           @logger.error("Couldn't find VM name in the output.")
185:           return nil
186:         end
187: 
188:         name = $1.to_s
189: 
190:         output = execute("list", "vms")
191:         if output =~ /^"#{Regexp.escape(name)}" \{(.+?)\}$/
192:           return $1.to_s
193:         end
194: 
195:         nil
196:       end

198:       def read_forwarded_ports(uuid=nil, active_only=false)
199:         uuid ||= @uuid
200: 
201:         @logger.debug("read_forward_ports: uuid=#{uuid} active_only=#{active_only}")
202: 
203:         results = []
204:         current_nic = nil
205:         execute("showvminfo", uuid, "--machinereadable").split("\n").each do |line|
206:           # This is how we find the nic that a FP is attached to,
207:           # since this comes first.
208:           current_nic = $1.to_i if line =~ /^nic(\d+)=".+?"$/
209: 
210:           # If we care about active VMs only, then we check the state
211:           # to verify the VM is running.
212:           if active_only && line =~ /^VMState="(.+?)"$/ && $1.to_s != "running"
213:             return []
214:           end
215: 
216:           # Parse out the forwarded port information
217:           if line =~ /^Forwarding.+?="(.+?),.+?,.*?,(.+?),.*?,(.+?)"$/
218:             result = [current_nic, $1.to_s, $2.to_i, $3.to_i]
219:             @logger.debug("  - #{result.inspect}")
220:             results << result
221:           end
222:         end
223: 
224:         results
225:       end
