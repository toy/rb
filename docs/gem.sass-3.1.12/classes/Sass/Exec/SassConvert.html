611:       def process_file(input, output)
612:         if input.is_a?(File)
613:           @options[:from] ||=
614:             case input.path
615:             when /\.scss$/; :scss
616:             when /\.sass$/; :sass
617:             when /\.less$/; :less
618:             when /\.css$/; :css
619:             end
620:         elsif @options[:in_place]
621:           raise "Error: the --in-place option requires a filename."
622:         end
623: 
624:         if output.is_a?(File)
625:           @options[:to] ||=
626:             case output.path
627:             when /\.scss$/; :scss
628:             when /\.sass$/; :sass
629:             end
630:         end
631: 
632:         @options[:from] ||= :css
633:         @options[:to] ||= :sass
634:         @options[:for_engine][:syntax] = @options[:from]
635: 
636:         out =
637:           ::Sass::Util.silence_sass_warnings do
638:             if @options[:from] == :css
639:               require 'sass/css'
640:               ::Sass::CSS.new(input.read, @options[:for_tree]).render(@options[:to])
641:             elsif @options[:from] == :less
642:               require 'sass/less'
643:               try_less_note
644:               input = input.read if input.is_a?(IO) && !input.is_a?(File) # Less is dumb
645:               Less::Engine.new(input).to_tree.to_sass_tree.send("to_#{@options[:to]}", @options[:for_tree])
646:             else
647:               if input.is_a?(File)
648:                 ::Sass::Engine.for_file(input.path, @options[:for_engine])
649:               else
650:                 ::Sass::Engine.new(input.read, @options[:for_engine])
651:               end.to_tree.send("to_#{@options[:to]}", @options[:for_tree])
652:             end
653:           end
654: 
655:         output = File.open(input.path, 'w') if @options[:in_place]
656:         output.write(out)
657:       rescue ::Sass::SyntaxError => e
658:         raise e if @options[:trace]
659:         file = " of #{e.sass_filename}" if e.sass_filename
660:         raise "Error on line #{e.sass_line}#{file}: #{e.message}\n  Use --trace for backtrace"
