47:         if !@exposed
48:           unless arguments.Length() == 1 && arguments[0].kind_of?(C::External)
49:             C::ThrowException(C::Exception::Error(C::String::New("cannot call native constructor from javascript")))

53:           end
54:         else

57:             instance = arguments[0].Value()
