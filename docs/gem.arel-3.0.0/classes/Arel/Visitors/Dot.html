230:       def to_dot
231:         "digraph \"ARel\" {\nnode [width=0.375,height=0.25,shape=record];\n" +
232:           @nodes.map { |node|
233:             label = "<f0>#{node.name}"

240:           }.join("\n") + "\n" + @edges.map { |edge|
241:             "#{edge.from.id} -> #{edge.to.id} [label=\"#{edge.name}\"];"
242:           }.join("\n") + "\n}"
243:       end
