137:         def feed(aBuf, aLen)
138:             if @_mDone then return end
139: 
140:             # The buffer we got is byte oriented, and a character may span in more than one
141:             # buffers. In case the last one or two byte in last buffer is not complete, we 
142:             # record how many byte needed to complete that character and skip these bytes here.
143:             # We can choose to record those bytes as well and analyse the character once it 
144:             # is complete, but since a character will not make much difference, by simply skipping
145:             # this character will simply our logic and improve performance.
146:             i = @_mNeedToSkipCharNum
147:             while i < aLen
148:                 order, charLen = get_order(aBuf[i..i+2])
149:                 i += charLen
150:                 if i > aLen
151:                     @_mNeedToSkipCharNum = i - aLen
152:                     @_mLastCharOrder = -1
153:                 else
154:                     if (order != -1) and (@_mLastCharOrder != -1)
155:                         @_mTotalRel += 1
156:                         if @_mTotalRel > MAX_REL_THRESHOLD
157:                             @_mDone = true
158:                             break
159:                         end
160:                         @_mRelSample[Jp2CharContext[@_mLastCharOrder][order]] += 1
161:                     end
162:                     @_mLastCharOrder = order
163:                 end
164:             end
165:         end
