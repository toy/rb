converge_locked_specs()

Remove elements from the locked specs that are expired. This will most commonly happen if the Gemfile has changed since the lockfile was last generated

Source: show 

     # File lib/bundler/definition.rb, line 355
355:     def converge_locked_specs
356:       deps = []
357: 
358:       # Build a list of dependencies that are the same in the Gemfile
359:       # and Gemfile.lock. If the Gemfile modified a dependency, but
360:       # the gem in the Gemfile.lock still satisfies it, this is fine
361:       # too.
362:       @dependencies.each do |dep|
363:         locked_dep = @locked_deps.find { |d| dep == d }
364: 
365:         if in_locked_deps?(dep, locked_dep) || satisfies_locked_spec?(dep)
366:           deps << dep
367:         elsif dep.source.is_a?(Source::Path) && dep.current_platform? && (!locked_dep || dep.source != locked_dep.source)
368:           @locked_specs.each do |s|
369:             @unlock[:gems] << s.name if s.source == dep.source
370:           end
371: 
372:           dep.source.unlock! if dep.source.respond_to?(:unlock!)
373:           dep.source.specs.each { |s| @unlock[:gems] << s.name }
374:         end
375:       end
376: 
377:       converged = []
378:       @locked_specs.each do |s|
379:         s.source = @sources.find { |src| s.source == src }
380: 
381:         # Don't add a spec to the list if its source is expired. For example,
382:         # if you change a Git gem to Rubygems.
383:         next if s.source.nil? || @unlock[:sources].include?(s.name)
384:         # If the spec is from a path source and it doesn't exist anymore
385:         # then we just unlock it.
386: 
387:         # Path sources have special logic
388:         if s.source.instance_of?(Source::Path)
389:           other = s.source.specs[s].first
390: 
391:           # If the spec is no longer in the path source, unlock it. This
392:           # commonly happens if the version changed in the gemspec
393:           next unless other
394: 
395:           deps2 = other.dependencies.select { |d| d.type != :development }
396:           # If the dependencies of the path source have changed, unlock it
397:           next unless s.dependencies.sort == deps2.sort
398:         end
399: 
400:         converged << s
401:       end
402: 
403:       resolve = SpecSet.new(converged)
404:       resolve = resolve.for(expand_dependencies(deps, true), @unlock[:gems])
405:       diff    = @locked_specs.to_a - resolve.to_a
406: 
407:       # Now, we unlock any sources that do not have anymore gems pinned to it
408:       @sources.each do |source|
409:         next unless source.respond_to?(:unlock!)
410: 
411:         unless resolve.any? { |s| s.source == source }
412:           source.unlock! if !diff.empty? && diff.any? { |s| s.source == source }
413:         end
414:       end
415: 
416:       resolve
417:     end
